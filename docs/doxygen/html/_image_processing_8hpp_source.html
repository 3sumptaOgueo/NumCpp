738  // Description:
741  void runClusterMaker()
742  {
743  uint32 clusterId = 0;
744 
745  for (uint32 xcdIdx = 0; xcdIdx < xcdsVec_.size(); ++xcdIdx)
746  {
747  Pixel<dtype>& currentPixel = xcdsVec_[xcdIdx];
748 
749  // not already visited
750  if (currentPixel.clusterId() == -1)
751  {
752  Cluster<dtype> newCluster(clusterId); // a new cluster
753  currentPixel.setClusterId(clusterId);
754  newCluster.addPixel(currentPixel); // assign pixel to cluster
755 
756  // get the neighbors
757  std::vector<uint32> neighborIds;
758  findNeighborXcds(currentPixel, neighborIds);
759  if (neighborIds.empty())
760  {
761  clusters_.push_back(newCluster);
762  ++clusterId;
763  continue;
764  }
765 
766  // loop through the neighbors
767  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
768  {
769  Pixel<dtype>& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
770 
771  // go to neighbors
772  std::vector<uint32> newNeighborIds;
773  findNeighborXcds(currentNeighborPixel, newNeighborIds);
774 
775  // loop through the new neighbors and add them to neighbors
776  for (uint32 newNeighborsIdx = 0; newNeighborsIdx < newNeighborIds.size(); ++newNeighborsIdx)
777  {
778  // not already in neighbors
779  if (find(neighborIds.begin(), neighborIds.end(), newNeighborIds[newNeighborsIdx]) == neighborIds.end())
780  {
781  neighborIds.push_back(newNeighborIds[newNeighborsIdx]);
782  }
783  }
784 
785  // not already assigned to a cluster
786  if (currentNeighborPixel.clusterId() == -1)
787  {
788  currentNeighborPixel.setClusterId(clusterId);
789  newCluster.addPixel(currentNeighborPixel);
790  }
791  }
792 
793  clusters_.push_back(newCluster);
794  ++clusterId;
795  }
796  }

972  void centerOfMass(const Cluster<dtype>& inCluster)
973  {
974  Shape clusterShape(inCluster.height(), inCluster.width());
975  NdArray<dtype> clusterArray(clusterShape);
976  clusterArray.zeros();
977 
978  uint32 rowMin = inCluster.rowMin();
979  uint32 colMin = inCluster.colMin();
980  dtype inten = inCluster.intensity();
981 
982  auto iter = inCluster.begin();
983  for (; iter < inCluster.end(); ++iter)
984  {
985  clusterArray(iter->row() - rowMin, iter->col() - colMin) = iter->intensity();
986  }
987 
988  // first get the row center
989  row_ = 0;
990  uint32 theRow = rowMin;
991  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
992  {
993  double rowSum = 0;
994  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
995  {
996  rowSum += static_cast<double>(clusterArray(rowIdx, colIdx));
997  }
998  row_ += rowSum * static_cast<double>(theRow++);
999  }
1000 
1001  row_ /= static_cast<double>(inten);
1002 
1003  // then get the column center
1004  col_ = 0;
1005  uint32 theCol = colMin;
1006  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
1007  {
1008  double colSum = 0;
1009  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
1010  {
1011  colSum += static_cast<double>(clusterArray(rowIdx, colIdx));
1012  }
1013  col_ += colSum * static_cast<double>(theCol++);
1014  }
1015 
1016  col_ /= static_cast<double>(inten);
1017  }

1309  template<typename dtype>
1310  dtype generateThreshold(const NdArray<dtype>& inImageArray, double inRate)
1311  {
1312  if (inRate < 0.0 || inRate > 1.0)
1313  {
1314  std::string errStr = "ERROR: ImageProcessing::generateThreshold: input rate must be of the range [0, 1]";
1315  std::cerr << errStr << std::endl;
1316  throw std::invalid_argument(errStr);
1317  }
1318 
1319  // first build a histogram
1320  int32 minValue = static_cast<int32>(std::floor(inImageArray.min().item()));
1321  int32 maxValue = static_cast<int32>(std::floor(inImageArray.max().item()));
1322 
1323  if (Utils::essentiallyEqual(inRate, 0.0))
1324  {
1325  return static_cast<dtype>(maxValue);
1326  }
1327  else if (Utils::essentiallyEqual(inRate, 1.0))
1328  {
1329  if (DtypeInfo<dtype>::isSigned())
1330  {
1331  return static_cast<dtype>(minValue - 1);
1332  }
1333  else
1334  {
1335  return static_cast<dtype>(0);
1336  }
1337  }
1338 
1339  uint32 histSize = static_cast<uint32>(maxValue - minValue + 1);
1340 
1341  NdArray<double> histogram(1, histSize);
1342  histogram.zeros();
1343  uint32 numPixels = inImageArray.size();
1344  for (uint32 i = 0; i < numPixels; ++i)
1345  {
1346  uint32 bin = static_cast<uint32>(static_cast<int32>(std::floor(inImageArray[i])) - minValue);
1347  ++histogram[bin];
1348  }
1349 
1350  // integrate the normalized histogram from right to left to make a survival function (1 - CDF)
1351  double dNumPixels = static_cast<double>(numPixels);
1352  NdArray<double> survivalFunction(1, histSize + 1);
1353  survivalFunction[-1] = 0;
1354  for (int32 i = histSize - 1; i > -1; --i)
1355  {
1356  double histValue = histogram[i] / dNumPixels;
1357  survivalFunction[i] = survivalFunction[i + 1] + histValue;
1358  }
1359 
1360  // binary search through the survival function to find the rate
1361  uint32 indexLow = 0;
1362  uint32 indexHigh = histSize - 1;
1363  uint32 index = indexHigh / 2; // integer division
1364 
1365  bool keepGoing = true;
1366  while (keepGoing)
1367  {
1368  double value = survivalFunction[index];
1369  if (value < inRate)
1370  {
1371  indexHigh = index;
1372  }
1373  else if (value > inRate)
1374  {
1375  indexLow = index;
1376  }
1377  else
1378  {
1379  int32 thresh = static_cast<int32>(index) + minValue - 1;
1380  if (DtypeInfo<dtype>::isSigned())
1381  {
1382  return static_cast<dtype>(thresh);
1383  }
1384  else
1385  {
1386  return thresh < 0 ? 0 : static_cast<dtype>(thresh);
1387  }
1388  }
1389 
1390  if (indexHigh - indexLow < 2)
1391  {
1392  return static_cast<dtype>(static_cast<int32>(indexHigh) + minValue - 1);
1393  }
1394 
1395  index = indexLow + (indexHigh - indexLow) / 2;
1396  }
1397 
1398  // shouldn't ever get here but stop the compiler from throwing a warning
1399  return static_cast<dtype>(histSize - 1);
1400  }
