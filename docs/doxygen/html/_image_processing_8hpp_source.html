831  // Description:
840  void runClusterMaker()
841  {
842  uint32 clusterId = 0;
843 
844  for (uint32 xcdIdx = 0; xcdIdx < xcdsVec_.size(); ++xcdIdx)
845  {
846  Pixel& currentPixel = xcdsVec_[xcdIdx];
847 
848  // not already visited
849  if (currentPixel.clusterId() == -1)
850  {
851  Cluster newCluster(clusterId); // a new cluster
852  currentPixel.setClusterId(clusterId);
853  newCluster.addPixel(currentPixel); // assign pixel to cluster
854 
855  // get the neighbors
856  std::vector<uint32> neighborIds;
857  findNeighborXcds(currentPixel, neighborIds);
858  if (neighborIds.empty())
859  {
860  clusters_.push_back(newCluster);
861  ++clusterId;
862  continue;
863  }
864 
865  // loop through the neighbors
866  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
867  {
868  Pixel& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
869 
870  // go to neighbors
871  std::vector<uint32> newNeighborIds;
872  findNeighborXcds(currentNeighborPixel, newNeighborIds);
873 
874  // loop through the new neighbors and add them to neighbors
875  for (uint32 newNeighborsIdx = 0; newNeighborsIdx < newNeighborIds.size(); ++newNeighborsIdx)
876  {
877  // not already in neighbors
878  if (find(neighborIds.begin(), neighborIds.end(), newNeighborIds[newNeighborsIdx]) == neighborIds.end())
879  {
880  neighborIds.push_back(newNeighborIds[newNeighborsIdx]);
881  }
882  }
883 
884  // not already assigned to a cluster
885  if (currentNeighborPixel.clusterId() == -1)
886  {
887  currentNeighborPixel.setClusterId(clusterId);
888  newCluster.addPixel(currentNeighborPixel);
889  }
890  }
891 
892  clusters_.push_back(newCluster);
893  ++clusterId;
894  }
895  }

1441  static dtype generateThreshold(const NdArray<dtype>& inImageArray, double inRate)
1442  {
1443  if (inRate < 0 || inRate > 1)
1444  {
1445  throw std::invalid_argument("ERROR: ImageProcessing::generateThreshold: input rate must be of the range [0, 1]");
1446  }
1447 
1448  // first build a histogram
1449  int32 minValue = static_cast<int32>(std::floor(inImageArray.min().item()));
1450  int32 maxValue = static_cast<int32>(std::floor(inImageArray.max().item()));
1451 
1452  if (inRate == 0)
1453  {
1454  return static_cast<dtype>(maxValue);
1455  }
1456  else if (inRate == 1)
1457  {
1458  if (DtypeInfo<dtype>::isSigned())
1459  {
1460  return static_cast<dtype>(minValue - 1);
1461  }
1462  else
1463  {
1464  return static_cast<dtype>(0);
1465  }
1466  }
1467 
1468  uint32 histSize = static_cast<uint32>(maxValue - minValue + 1);
1469 
1470  NdArray<double> histogram(1, histSize);
1471  histogram.zeros();
1472  uint32 numPixels = inImageArray.size();
1473  for (uint32 i = 0; i < numPixels; ++i)
1474  {
1475  uint32 bin = static_cast<uint32>(static_cast<int32>(std::floor(inImageArray[i])) - minValue);
1476  ++histogram[bin];
1477  }
1478 
1479  // integrate the normalized histogram from right to left to make a survival function (1 - CDF)
1480  double dNumPixels = static_cast<double>(numPixels);
1481  NdArray<double> survivalFunction(1, histSize + 1);
1482  survivalFunction[-1] = 0;
1483  for (int32 i = histSize - 1; i > -1; --i)
1484  {
1485  double histValue = histogram[i] / dNumPixels;
1486  survivalFunction[i] = survivalFunction[i + 1] + histValue;
1487  }
1488 
1489  // binary search through the survival function to find the rate
1490  uint32 indexLow = 0;
1491  uint32 indexHigh = histSize - 1;
1492  uint32 index = indexHigh / 2; // integer division
1493 
1494  bool keepGoing = true;
1495  while (keepGoing)
1496  {
1497  double value = survivalFunction[index];
1498  if (value < inRate)
1499  {
1500  indexHigh = index;
1501  }
1502  else if (value > inRate)
1503  {
1504  indexLow = index;
1505  }
1506  else
1507  {
1508  int32 thresh = static_cast<int32>(index) + minValue - 1;
1509  if (DtypeInfo<dtype>::isSigned())
1510  {
1511  return static_cast<dtype>(thresh);
1512  }
1513  else
1514  {
1515  return thresh < 0 ? 0 : static_cast<dtype>(thresh);
1516  }
1517  }
1518 
1519  if (indexHigh - indexLow < 2)
1520  {
1521  return static_cast<dtype>(static_cast<int32>(indexHigh) + minValue - 1);
1522  }
1523 
1524  index = indexLow + (indexHigh - indexLow) / 2;
1525  }
1526 
1527  // shouldn't ever get here but stop the compiler from throwing a warning
1528  return static_cast<dtype>(histSize - 1);
1529  }
