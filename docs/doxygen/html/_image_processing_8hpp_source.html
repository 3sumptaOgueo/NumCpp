748  currentPixel.setClusterId(clusterId);
749  newCluster.addPixel(currentPixel); // assign pixel to cluster
750

752  std::vector<uint32> neighborIds;
753  findNeighborXcds(currentPixel, neighborIds);
754  if (neighborIds.empty())
755  {
756  clusters_.push_back(newCluster);
757  ++clusterId;
758  continue;
759  }
760 
761  // loop through the neighbors
762  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
763  {
764  Pixel& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
765 
766  // go to neighbors
767  std::vector<uint32> newNeighborIds;
768  findNeighborXcds(currentNeighborPixel, newNeighborIds);
769 
770  // loop through the new neighbors and add them to neighbors
771  for (uint32 newNeighborsIdx = 0; newNeighborsIdx < newNeighborIds.size(); ++newNeighborsIdx)
772  {
773  // not already in neighbors
774  if (find(neighborIds.begin(), neighborIds.end(), newNeighborIds[newNeighborsIdx]) == neighborIds.end())
775  {
776  neighborIds.push_back(newNeighborIds[newNeighborsIdx]);
777  }
778  }
779 
780  // not already assigned to a cluster
781  if (currentNeighborPixel.clusterId() == -1)
782  {
783  currentNeighborPixel.setClusterId(clusterId);
784  newCluster.addPixel(currentNeighborPixel);
785  }
786  }

993

997  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)

1000  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
1001  {
1002  colSum += static_cast<double>(clusterArray(rowIdx, colIdx));
1003  }

1005  }

1325  uint32 indexLow = 0;
1326  uint32 indexHigh = histSize - 1;

1328

1332  double value = survivalFunction[index];

1334  {
1335  indexHigh = index;
1336  }
1337  else if (value > inRate)
1338  {

1340  }
1341  else
