174  static NdArray<double> angularVelocity(const Quaternion& inQuat1, const Quaternion& inQuat2, double inTime)
175  {
176  NdArray<double> q0 = inQuat1.toNdArray();
177  NdArray<double> q1 = inQuat2.toNdArray();
178 
179  NdArray<double> qDot = q1 - q0;
180  qDot /= inTime;
181 
182  NdArray<double> eyeTimesScalar(3);
183  eyeTimesScalar.zeros();
184  eyeTimesScalar(0, 0) = inQuat2.s();
185  eyeTimesScalar(1, 1) = inQuat2.s();
186  eyeTimesScalar(2, 2) = inQuat2.s();
187 
188  NdArray<double> epsilonHat = Linalg<double>::hat(inQuat2.i(), inQuat2.j(), inQuat2.k());
189  NdArray<double> q(4, 3);
190  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
191  q(3, 0) = -inQuat2.i();
192  q(3, 1) = -inQuat2.j();
193  q(3, 2) = -inQuat2.k();
194 
195  NdArray<double> omega = q.transpose().dot<double>(qDot.transpose());
196  return std::move(omega *= 2);

309  static Quaternion fromDCM(const NdArray<dtype>& inDcm)
310  {
311  Shape inShape = inDcm.shape();
312  if (!(inShape.rows == 3 && inShape.cols == 3))
313  {
314  throw std::invalid_argument("ERROR: Rotations::Quaternion::fromDcm: input direction cosine matrix must have shape = (3,3).");
315  }
316 
317  NdArray<double> dcm = inDcm.astype();
318 
319  NdArray<double> checks(1, 4);
320  checks[0] = dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
321  checks[1] = dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
322  checks[2] = dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
323  checks[3] = dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
324 
325  uint32 maxIdx = Methods<double>::argmax(checks).item();
326 
327  double q0 = 0;
328  double q1 = 0;
329  double q2 = 0;
330  double q3 = 0;
331 
332  switch (maxIdx)
333  {
334  case 0:
335  {
336  q3 = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
337  q0 = (dcm(1, 2) - dcm(2, 1)) / (4 * q3);
338  q1 = (dcm(2, 0) - dcm(0, 2)) / (4 * q3);
339  q2 = (dcm(0, 1) - dcm(1, 0)) / (4 * q3);
340 
341  break;
342  }
343  case 1:
344  {
345  q2 = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
346  q0 = (dcm(0, 2) + dcm(2, 0)) / (4 * q2);
347  q1 = (dcm(1, 2) + dcm(2, 1)) / (4 * q2);
348  q3 = (dcm(0, 1) - dcm(1, 0)) / (4 * q2);
349 
350  break;
351  }
352  case 2:
353  {
354  q0 = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
355  q1 = (dcm(0, 1) + dcm(1, 0)) / (4 * q0);
356  q2 = (dcm(0, 2) + dcm(2, 0)) / (4 * q0);
357  q3 = (dcm(1, 2) - dcm(2, 1)) / (4 * q0);
358 
359  break;
360  }
361  case 3:
362  {
363  q1 = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
364  q0 = (dcm(0, 1) + dcm(1, 0)) / (4 * q1);
365  q2 = (dcm(1, 2) + dcm(2, 1)) / (4 * q1);
366  q3 = (dcm(2, 0) - dcm(0, 2)) / (4 * q1);
367 
368  break;
369  }
370  }
371 
372  return Quaternion(q0, q1, q2, q3);
373  }

482  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
483  {
484  if (inPercent < 0 || inPercent > 1)
485  {
486  throw std::invalid_argument("ERROR: Rotations::Quaternion::slerp: input percent must be of the range [0, 1]");
487  }
488 
489  if (inPercent == 0)
490  {
491  return inQuat1;
492  }
493  else if (inPercent == 1)
494  {
495  return inQuat2;
496  }
497 
498  double dotProduct = Methods<double>::dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
499 
500  // If the dot product is negative, the quaternions
501  // have opposite handed-ness and slerp won't take
502  // the shorter path. Fix by reversing one quaternion.
503  Quaternion quat1Copy(inQuat1);
504  if (dotProduct < 0.0)
505  {
506  quat1Copy *= -1;
507  dotProduct *= -1;
508  }
509 
510  const double DOT_THRESHOLD = 0.9995;
511  if (dotProduct > DOT_THRESHOLD) {
512  // If the inputs are too close for comfort, linearly interpolate
513  // and normalize the result.
514  return nlerp(inQuat1, inQuat2, inPercent);
515  }
516 
517  dotProduct = Methods<double>::clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
518  double theta0 = std::acos(dotProduct); // angle between input vectors
519  double theta = theta0 * inPercent; // angle between v0 and result
520 
521  double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
522  double s1 = std::sin(theta) / std::sin(theta0);
523 
524  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
525  return Quaternion(interpQuat);
526  }

568  NdArray<double> toDCM() const
569  {
570  NdArray<double> dcm(3);
571 
572  double q0 = i();
573  double q1 = j();
574  double q2 = k();
575  double q3 = s();
576 
577  dcm(0, 0) = Utils<double>::sqr(q3) + Utils<double>::sqr(q0) - Utils<double>::sqr(q1) - Utils<double>::sqr(q2);
578  dcm(0, 1) = 2 * (q0 * q1 + q3 * q2);
579  dcm(0, 2) = 2 * (q0 * q2 - q3 * q1);
580  dcm(1, 0) = 2 * (q0 * q1 - q3 * q2);
581  dcm(1, 1) = Utils<double>::sqr(q3) - Utils<double>::sqr(q0) + Utils<double>::sqr(q1) - Utils<double>::sqr(q2);;
582  dcm(1, 2) = 2 * (q1 * q2 + q3 * q0);
583  dcm(2, 0) = 2 * (q0 * q2 + q3 * q1);
584  dcm(2, 1) = 2 * (q1 * q2 - q3 * q0);
585  dcm(2, 2) = Utils<double>::sqr(q3) - Utils<double>::sqr(q0) - Utils<double>::sqr(q1) + Utils<double>::sqr(q2);;
586
