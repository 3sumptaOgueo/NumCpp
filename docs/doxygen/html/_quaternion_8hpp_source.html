303  {
304  const Shape inShape = inDcm.shape();

306  {
307  THROW_INVALID_ARGUMENT_ERROR("input direction cosine matrix must have shape = (3,3).");
308  }
309 
310  NdArray<double> checks(1, 4);
311  checks[0] = 1 + inDcm(0, 0) + inDcm(1, 1) + inDcm(2, 2);
312  checks[1] = 1 + inDcm(0, 0) - inDcm(1, 1) - inDcm(2, 2);
313  checks[2] = 1 - inDcm(0, 0) + inDcm(1, 1) - inDcm(2, 2);
314  checks[3] = 1 - inDcm(0, 0) - inDcm(1, 1) + inDcm(2, 2);
315 
316  const uint32 maxIdx = argmax(checks).item();
317 
318  double q0 = 0;
319  double q1 = 0;
320  double q2 = 0;
321  double q3 = 0;
322 
323  switch (maxIdx)
324  {
325  case 0:
326  {
327  q3 = 0.5 * std::sqrt(1 + inDcm(0, 0) + inDcm(1, 1) + inDcm(2, 2));
328  q0 = (inDcm(2, 1) - inDcm(1, 2)) / (4 * q3);
329  q1 = (inDcm(0, 2) - inDcm(2, 0)) / (4 * q3);
330  q2 = (inDcm(1, 0) - inDcm(0, 1)) / (4 * q3);
331 
332  break;
333  }
334  case 1:
335  {
336  q0 = 0.5 * std::sqrt(1 + inDcm(0, 0) - inDcm(1, 1) - inDcm(2, 2));
337  q1 = (inDcm(1, 0) + inDcm(0, 1)) / (4 * q0);
338  q2 = (inDcm(2, 0) + inDcm(0, 2)) / (4 * q0);
339  q3 = (inDcm(2, 1) - inDcm(1, 2)) / (4 * q0);
340 
341  break;
342  }
343  case 2:

345  q1 = 0.5 * std::sqrt(1 - inDcm(0, 0) + inDcm(1, 1) - inDcm(2, 2));
346  q0 = (inDcm(1, 0) + inDcm(0, 1)) / (4 * q1);
347  q2 = (inDcm(2, 1) + inDcm(1, 2)) / (4 * q1);
348  q3 = (inDcm(0, 2) - inDcm(2, 0)) / (4 * q1);
349

479  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
480  {
481  if (inPercent < 0 || inPercent > 1)
482  {
483  THROW_INVALID_ARGUMENT_ERROR("input percent must be of the range [0, 1]");
484  }
485 
486  if (inPercent == 0)
487  {
488  return inQuat1;
489  }
490  else if (inPercent == 1)
491  {
492  return inQuat2;
493  }
494 
495  double dotProduct = dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
496 
497  // If the dot product is negative, the quaternions
498  // have opposite handed-ness and slerp won't take
499  // the shorter path. Fix by reversing one quaternion.
500  Quaternion quat1Copy(inQuat1);
501  if (dotProduct < 0.0)
502  {
503  quat1Copy *= -1;
504  dotProduct *= -1;
505  }
506 
507  const double DOT_THRESHOLD = 0.9995;
508  if (dotProduct > DOT_THRESHOLD) {
509  // If the inputs are too close for comfort, linearly interpolate
510  // and normalize the result.
511  return nlerp(inQuat1, inQuat2, inPercent);
512  }
513 
514  dotProduct = clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
515  const double theta0 = std::acos(dotProduct); // angle between input vectors
516  const double theta = theta0 * inPercent; // angle between v0 and result
517 
518  const double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
519  const double s1 = std::sin(theta) / std::sin(theta0);
520 
521  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
522  return Quaternion(interpQuat);
523  }

581  dcm(2, 0) = 2 * (q0 * q2 - q3 * q1);
582  dcm(2, 1) = 2 * (q1 * q2 + q3 * q0);
583  dcm(2, 2) = q3sqr + q2sqr - q0sqr - q1sqr;
