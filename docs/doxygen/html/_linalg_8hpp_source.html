82  dtype cdh = inArray(0, 2) * inArray(1, 0) * inArray(2, 1);
83  dtype ceg = inArray(0, 2) * inArray(1, 1) * inArray(2, 0);
84  dtype bdi = inArray(0, 1) * inArray(1, 0) * inArray(2, 2);
85  dtype afh = inArray(0, 0) * inArray(1, 2) * inArray(2, 1);
86 
87  return aei + bfg + cdh - ceg - bdi - afh;
88  }
89  else
90  {
91  dtype determinant = 0;
92  NdArray<dtype> submat(inShape.rows - 1);
93 
94  for (uint32 c = 0; c < inShape.rows; ++c)
95  {
96  uint32 subi = 0;

98  {
99  uint32 subj = 0;
