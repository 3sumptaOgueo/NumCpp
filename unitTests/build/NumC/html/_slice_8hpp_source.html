151  // Method Description:
152  // Make the slice all positive and does some error checking
153  //
155  // The calling array size
157  // None
158  //
159  void makePositiveAndValidate(uint32 inArraySize)
160  {
161  // convert the start value
162  if (start < 0)
163  {
164  start += inArraySize;
165  }
166  if (start > static_cast<int32>(inArraySize - 1))
167  {
168  throw std::invalid_argument("ERROR: Invalid start value for array of size " + Utils<uint32>::num2str(inArraySize) + ".");
169  }
170 
171  // convert the stop value
172  if (stop < 0)
173  {
174  stop += inArraySize;
175  }
176  if (stop > static_cast<int32>(inArraySize))
177  {
178  throw std::invalid_argument("ERROR: Invalid stop value for array of size " + Utils<uint32>::num2str(inArraySize) + ".");
179  }
180 
181  // do some error checking
182  if (start < stop)
183  {
184  if (step < 0)
185  {
186  throw std::invalid_argument("ERROR: Invalid slice values.");
187  }
188  }
189 
190  if (stop < start)
191  {
192  if (step > 0)
193  {
194  throw std::invalid_argument("ERROR: Invalid slice values.");
195  }
196 
197  // otherwise flip things around for my own sanity
198  std::swap(start, stop);
199  step *= -1;
200  }
201  }
202 
203  //============================================================================
204  // Method Description:

206  // be aware that this method will also make the slice all
207  // positive!
208  //
210  // The calling array size

214  uint32 numElements(uint32 inArraySize)
215  {

222  }
