795  //=============================================================================
796  // Description:
805  void findNeighborXcds(const Pixel& inPixel, std::vector<uint32>& outNeighbors)
806  {
807  std::set<Pixel> neighbors;
808  findNeighbors(inPixel, neighbors);
809  std::vector<Pixel> neighborXcds;
810 
811  // check if the neighboring pixels are exceedances and insert into the xcd vector
812  for (auto pixelIter = neighbors.begin(); pixelIter != neighbors.end(); ++pixelIter)
813  {
814  if (xcds_->operator()(pixelIter->row(), pixelIter->col()))
815  {
816  neighborXcds.push_back(*pixelIter);
817  }
818  }
819 
820  // loop through the neighbors and find the cooresponding index into exceedances_
821  for (auto pixelIter = neighborXcds.begin(); pixelIter < neighborXcds.end(); ++pixelIter)
822  {
823  auto theExceedanceIter = find(xcdsVec_.begin(), xcdsVec_.end(), *pixelIter);
824  outNeighbors.push_back(static_cast<uint32>(theExceedanceIter - xcdsVec_.begin()));
825  }

828  //=============================================================================
829  // Description:
838  void runClusterMaker()
839  {
840  uint32 clusterId = 0;
841 
842  for (uint32 xcdIdx = 0; xcdIdx < xcdsVec_.size(); ++xcdIdx)
843  {
844  Pixel& currentPixel = xcdsVec_[xcdIdx];
845 
846  // not already visited
847  if (currentPixel.clusterId() == -1)
848  {
849  Cluster newCluster(clusterId); // a new cluster
850  currentPixel.setClusterId(clusterId);
851  newCluster.addPixel(currentPixel); // assign pixel to cluster
852 
853  // get the neighbors
854  std::vector<uint32> neighborIds;
855  findNeighborXcds(currentPixel, neighborIds);
856  if (neighborIds.empty())
857  {
858  clusters_.push_back(newCluster);
859  ++clusterId;
860  continue;
861  }
862 
863  // loop through the neighbors
864  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
865  {
866  Pixel& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
867 
868  // go to neighbors
869  std::vector<uint32> newNeighborIds;
870  findNeighborXcds(currentNeighborPixel, newNeighborIds);
871 
872  // loop through the new neighbors and add them to neighbors
873  for (uint32 newNeighborsIdx = 0; newNeighborsIdx < newNeighborIds.size(); ++newNeighborsIdx)
874  {
875  // not already in neighbors
876  if (find(neighborIds.begin(), neighborIds.end(), newNeighborIds[newNeighborsIdx]) == neighborIds.end())
877  {
878  neighborIds.push_back(newNeighborIds[newNeighborsIdx]);
879  }
880  }
881 
882  // not already assigned to a cluster
883  if (currentNeighborPixel.clusterId() == -1)
884  {
885  currentNeighborPixel.setClusterId(clusterId);
886  newCluster.addPixel(currentNeighborPixel);
887  }
888  }
889 
890  clusters_.push_back(newCluster);
891  ++clusterId;
892  }
893  }
894  }

906  void expandClusters()
907  {
908  // loop through the clusters
909  for (auto clusterIter = clusters_.begin(); clusterIter < clusters_.end(); ++clusterIter)
910  {
911  // loop through the pixels of the cluster
912  Cluster& theCluster = *clusterIter;
913  uint32 clusterSize = static_cast<uint32>(theCluster.size());
914  for (uint32 iPixel = 0; iPixel < clusterSize; ++iPixel)
915  {

917  std::vector<Pixel> neighborsNotXcds;
918  findNeighborNotXcds(thePixel, neighborsNotXcds);
919 
920  // loop through the neighbors and if they haven't already been added to the cluster, add them
921  for (auto newPixelIter = neighborsNotXcds.begin(); newPixelIter < neighborsNotXcds.end(); ++newPixelIter)
922  {
923  if (find(theCluster.begin(), theCluster.end(), *newPixelIter) == theCluster.end())
924  {
925  theCluster.addPixel(*newPixelIter);
926  }
927  }
928  }

1083  void centerOfMass(const Cluster& inCluster)
1084  {
1085  Shape clusterShape(inCluster.height(), inCluster.width());
1086  NdArray<dtype> clusterArray(clusterShape);
1087  clusterArray.zeros();
1088 
1089  uint32 rowMin = inCluster.rowMin();
1090  uint32 colMin = inCluster.colMin();
1091  dtype intensity = inCluster.intensity();
1092 
1093  auto iter = inCluster.begin();
1094  for (; iter < inCluster.end(); ++iter)
1095  {
1096  clusterArray(iter->row() - rowMin, iter->col() - colMin) = iter->intensity();
1097  }
1098 
1099  // first get the row center
1100  row_ = 0;
1101  uint32 row = rowMin;
1102  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
1103  {
1104  double rowSum = 0;
1105  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
1106  {
1107  rowSum += static_cast<double>(clusterArray(rowIdx, colIdx));
1108  }
1109  row_ += rowSum * static_cast<double>(row++);
1110  }
1111 
1112  row_ /= static_cast<double>(intensity);
1113 
1114  // then get the column center
1115  col_ = 0;
1116  uint32 col = colMin;
1117  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
1118  {
1119  double colSum = 0;
1120  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
1121  {
1122  colSum += static_cast<double>(clusterArray(rowIdx, colIdx));
1123  }
1124  col_ += colSum * static_cast<double>(col++);
1125  }
1126 
1127  col_ /= static_cast<double>(intensity);
1128  }

1503  }
